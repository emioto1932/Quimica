// Configuração do canvas e membrana const canvasWidth = 500; // Largura do canvas const canvasHeight = 500; // Altura do canvas const thickness = 20; // Espessura da membrana const spacing = 15; // Espaçamento entre as bolinhas // Coordenadas da membrana const membraneYTop = (canvasHeight / 2) - (thickness / 2); // Parte superior da membrana const membraneYBottom = (canvasHeight / 2) + (thickness / 2); // Parte inferior da membrana // Bolinhas let extracellularNa = []; // Bolinhas de Na+ na região extracelular let extracellularK = []; // Bolinhas de K+ na região extracelular let intracellularNa = []; // Bolinhas de Na+ na região intracelular let intracellularK = []; // Bolinhas de K+ na região intracelular // Função para criar bolinhas em uma posição inicial function createBolinhas() { // Cria bolinhas extracelulares (alternando Na+ e K+) for (let i = 0; i < 10; i++) { extracellularNa.push({ x: i * spacing, y: membraneYTop }); } for (let i = 0; i < 9; i++) { extracellularK.push({ x: i * spacing + spacing / 2, y: membraneYTop }); } // Cria bolinhas intracelulares (alternando Na+ e K+) for (let i = 0; i < 6; i++) { intracellularNa.push({ x: i * spacing, y: membraneYBottom }); } for (let i = 0; i < 3; i++) { intracellularK.push({ x: i * spacing + spacing / 2, y: membraneYBottom }); } } // Função para desenhar as bolinhas function drawBolinhas(ctx, bolinhas, color, label) { ctx.fillStyle = color; ctx.font = "12px Arial"; ctx.textAlign = "center"; ctx.textBaseline = "middle"; bolinhas.forEach(bolinha => { ctx.beginPath(); ctx.arc(bolinha.x, bolinha.y, 8, 0, Math.PI * 2); // Desenha a bolinha ctx.fill(); ctx.fillStyle = "black"; // Cor do texto ctx.fillText(label, bolinha.x, bolinha.y); // Adiciona o texto no centro da bolinha }); } // Função principal para desenhar tudo function draw() { const canvas = document.getElementById("canvas"); const ctx = canvas.getContext("2d"); // Limpa o canvas ctx.clearRect(0, 0, canvasWidth, canvasHeight); // Desenha a membrana drawMembrane(ctx); // Desenha as bolinhas drawBolinhas(ctx, extracellularNa, "red", "Na+"); drawBolinhas(ctx, extracellularK, "blue", "K+"); drawBolinhas(ctx, intracellularNa, "red", "Na+"); drawBolinhas(ctx, intracellularK, "blue", "K+"); } // Configuração inicial function init() { createBolinhas(); draw(); } // Chama a função inicial init();
